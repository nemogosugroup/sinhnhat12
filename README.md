# Installation 
- Node version: >= 16.20.2
- Php version: >= 7.3.33
- Laravel version: 8
- Composer install
- npm install
# run
- php artisan migrate
- php artisan db:seed
- php artisan vendor:publish --tag=translatable
- npm run watch
- npm run dev
- npm run prod
- php artisan serve
